def get_user_nums():
    nums = []
    for i in range(5):
        num = input("Enter an int (or 'q' to exit): ")
        if num == "q":
            break
        nums.append(int(num))
    return nums

total = 0
for num in get_user_nums():
    total += num
    print("Total:", total)
